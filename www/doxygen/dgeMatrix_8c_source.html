342 {
343  int lg = asLogical(logarithm);
344  int *dims = INTEGER(GET_SLOT(x, Matrix_DimSym)),
345  n = dims[0], sign = 1;
346  double modulus = lg ? 0. : 1; /* initialize; = result for n == 0 */
347 
348  if (n != dims[1])
349  error(_("Determinant requires a square matrix"));
350  if (n > 0) {
351  SEXP lu = dgeMatrix_LU_(x, /* do not warn about singular LU: */ FALSE);
352  int i, *jpvt = INTEGER(GET_SLOT(lu, Matrix_permSym));
353  double *luvals = REAL(GET_SLOT(lu, Matrix_xSym));
354 
355  for (i = 0; i < n; i++) if (jpvt[i] != (i + 1)) sign = -sign;
356  if (lg) {
357  for (i = 0; i < n; i++) {
358  double dii = luvals[i*(n + 1)]; /* ith diagonal element */
359  modulus += log(dii < 0 ? -dii : dii);
360  if (dii < 0) sign = -sign;
361  }
362  } else {
363  for (i = 0; i < n; i++)
364  modulus *= luvals[i*(n + 1)];
365  if (modulus < 0) {
366  modulus = -modulus;
367  sign = -sign;
368  }
369  }
370  }
371  return as_det_obj(modulus, lg, sign);
372 }

374 SEXP dgeMatrix_solve(SEXP a)
375 {
376  /* compute the 1-norm of the matrix, which is needed
377  later for the computation of the reciprocal condition number. */
378  double aNorm = get_norm(a, "1");
379 
380  /* the LU decomposition : */
381  SEXP val = PROTECT(NEW_OBJECT(MAKE_CLASS("dgeMatrix"))),
382  lu = dgeMatrix_LU_(a, TRUE);
383  int *dims = INTEGER(GET_SLOT(lu, Matrix_DimSym)),
384  *pivot = INTEGER(GET_SLOT(lu, Matrix_permSym));
385 
386  /* prepare variables for the dgetri calls */
387  double *x, tmp;
388  int info, lwork = -1;
389 
390 
391  if (dims[0] != dims[1]) error(_("Solve requires a square matrix"));
392  slot_dup(val, lu, Matrix_xSym);
393  x = REAL(GET_SLOT(val, Matrix_xSym));
394  slot_dup(val, lu, Matrix_DimSym);
395 
396  if(dims[0]) /* the dimension is not zero */
397  {
398  /* is the matrix is *computationally* singular ? */
399  double rcond;
400  F77_CALL(dgecon)("1", dims, x, dims, &aNorm, &rcond,
401  (double *) R_alloc(4*dims[0], sizeof(double)),
402  (int *) R_alloc(dims[0], sizeof(int)), &info);
403  if (info)
404  error(_("error [%d] from Lapack 'dgecon()'"), info);
405  if(rcond < DOUBLE_EPS)
406  error(_("Lapack dgecon(): system computationally singular, reciprocal condition number = %g"),
407  rcond);
408 
409  /* only now try the inversion and check if the matrix is *exactly* singular: */
410  F77_CALL(dgetri)(dims, x, dims, pivot, &tmp, &lwork, &info);
411  lwork = (int) tmp;
412  F77_CALL(dgetri)(dims, x, dims, pivot,
413  (double *) R_alloc((size_t) lwork, sizeof(double)),
414  &lwork, &info);
415  if (info)
416  error(_("Lapack routine dgetri: system is exactly singular"));
417  }
418  UNPROTECT(1);
419  return val;
420 }

557 SEXP dgeMatrix_exp(SEXP x)
558 {
559  const double one = 1.0, zero = 0.0;
560  const int i1 = 1;
561  int *Dims = INTEGER(GET_SLOT(x, Matrix_DimSym));
562  const int n = Dims[1], nsqr = n * n, np1 = n + 1;
563 
564  SEXP val = PROTECT(duplicate(x));
565  int i, ilo, ilos, ihi, ihis, j, sqpow;
566  int *pivot = Calloc(n, int);
567  double *dpp = Calloc(nsqr, double), /* denominator power Pade' */
568  *npp = Calloc(nsqr, double), /* numerator power Pade' */
569  *perm = Calloc(n, double),
570  *scale = Calloc(n, double),
571  *v = REAL(GET_SLOT(val, Matrix_xSym)),
572  *work = Calloc(nsqr, double), inf_norm, m1_j/*= (-1)^j */, trshift;
573  R_CheckStack();
574 
575  if (n < 1 || Dims[0] != n)
576  error(_("Matrix exponential requires square, non-null matrix"));
577  if(n == 1) {
578  v[0] = exp(v[0]);
579  UNPROTECT(1);
580  return val;
581  }
582 
583  /* Preconditioning 1. Shift diagonal by average diagonal if positive. */
584  trshift = 0; /* determine average diagonal element */
585  for (i = 0; i < n; i++) trshift += v[i * np1];
586  trshift /= n;
587  if (trshift > 0.) { /* shift diagonal by -trshift */
588  for (i = 0; i < n; i++) v[i * np1] -= trshift;
589  }
590 
591  /* Preconditioning 2. Balancing with dgebal. */
592  F77_CALL(dgebal)("P", &n, v, &n, &ilo, &ihi, perm, &j);
593  if (j) error(_("dgeMatrix_exp: LAPACK routine dgebal returned %d"), j);
594  F77_CALL(dgebal)("S", &n, v, &n, &ilos, &ihis, scale, &j);
595  if (j) error(_("dgeMatrix_exp: LAPACK routine dgebal returned %d"), j);
596 
597  /* Preconditioning 3. Scaling according to infinity norm */
598  inf_norm = F77_CALL(dlange)("I", &n, &n, v, &n, work);
599  sqpow = (inf_norm > 0) ? (int) (1 + log(inf_norm)/log(2.)) : 0;
600  if (sqpow < 0) sqpow = 0;
601  if (sqpow > 0) {
602  double scale_factor = 1.0;
603  for (i = 0; i < sqpow; i++) scale_factor *= 2.;
604  for (i = 0; i < nsqr; i++) v[i] /= scale_factor;
605  }
606 
607  /* Pade' approximation. Powers v^8, v^7, ..., v^1 */
608  AZERO(npp, nsqr);
609  AZERO(dpp, nsqr);
610  m1_j = -1;
611  for (j = 7; j >=0; j--) {
612  double mult = padec[j];
613  /* npp = m * npp + padec[j] *m */
614  F77_CALL(dgemm)("N", "N", &n, &n, &n, &one, v, &n, npp, &n,
615  &zero, work, &n);
616  for (i = 0; i < nsqr; i++) npp[i] = work[i] + mult * v[i];
617  /* dpp = m * dpp + (m1_j * padec[j]) * m */
618  mult *= m1_j;
619  F77_CALL(dgemm)("N", "N", &n, &n, &n, &one, v, &n, dpp, &n,
620  &zero, work, &n);
621  for (i = 0; i < nsqr; i++) dpp[i] = work[i] + mult * v[i];
622  m1_j *= -1;
623  }
624  /* Zero power */
625  for (i = 0; i < nsqr; i++) dpp[i] *= -1.;
626  for (j = 0; j < n; j++) {
627  npp[j * np1] += 1.;
628  dpp[j * np1] += 1.;
629  }
630 
631  /* Pade' approximation is solve(dpp, npp) */
632  F77_CALL(dgetrf)(&n, &n, dpp, &n, pivot, &j);
633  if (j) error(_("dgeMatrix_exp: dgetrf returned error code %d"), j);
634  F77_CALL(dgetrs)("N", &n, &n, dpp, &n, pivot, npp, &n, &j);
635  if (j) error(_("dgeMatrix_exp: dgetrs returned error code %d"), j);
636  Memcpy(v, npp, nsqr);
637 
638  /* Now undo all of the preconditioning */
639  /* Preconditioning 3: square the result for every power of 2 */
640  while (sqpow--) {
641  F77_CALL(dgemm)("N", "N", &n, &n, &n, &one, v, &n, v, &n,
642  &zero, work, &n);
643  Memcpy(v, work, nsqr);
644  }
645  /* Preconditioning 2: apply inverse scaling */
646  for (j = 0; j < n; j++)
647  for (i = 0; i < n; i++)
648  v[i + j * n] *= scale[i]/scale[j];
649 
650 
651  /* 2 b) Inverse permutation (if not the identity permutation) */
652  if (ilo != 1 || ihi != n) {
653  /* Martin Maechler's code */
654 
655 #define SWAP_ROW(I,J) F77_CALL(dswap)(&n, &v[(I)], &n, &v[(J)], &n)
656 
657 #define SWAP_COL(I,J) F77_CALL(dswap)(&n, &v[(I)*n], &i1, &v[(J)*n], &i1)
658 
659 #define RE_PERMUTE(I) \
660  int p_I = (int) (perm[I]) - 1; \
661  SWAP_COL(I, p_I); \
662  SWAP_ROW(I, p_I)
663 
664  /* reversion of "leading permutations" : in reverse order */
665  for (i = (ilo - 1) - 1; i >= 0; i--) {
666  RE_PERMUTE(i);
667  }
668 
669  /* reversion of "trailing permutations" : applied in forward order */
670  for (i = (ihi + 1) - 1; i < n; i++) {
671  RE_PERMUTE(i);
672  }
673  }
674 
675  /* Preconditioning 1: Trace normalization */
676  if (trshift > 0.) {
677  double mult = exp(trshift);
678  for (i = 0; i < nsqr; i++) v[i] *= mult;
679  }
680 
681  /* Clean up */
682  Free(work); Free(scale); Free(perm); Free(npp); Free(dpp); Free(pivot);
683  UNPROTECT(1);
684  return val;
685 }
