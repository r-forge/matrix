70SEXP dgeMatrix_exp(SEXP x)
71{
72 const double one = 1.0, zero = 0.0;
73 const int i1 = 1;
74 int *Dims = INTEGER(GET_SLOT(x, Matrix_DimSym));
75 const int n = Dims[1];
76 const R_xlen_t n_ = n, np1 = n + 1, nsqr = n_ * n_; // nsqr = n^2
77
78 SEXP val = PROTECT(duplicate(x));
79 int i, ilo, ilos, ihi, ihis, j, sqpow;
80 int *pivot = R_Calloc(n, int);
81 double *dpp = R_Calloc(nsqr, double), /* denominator power Pade' */
82 *npp = R_Calloc(nsqr, double), /* numerator power Pade' */
83 *perm = R_Calloc(n, double),
84 *scale = R_Calloc(n, double),
85 *v = REAL(GET_SLOT(val, Matrix_xSym)),
86 *work = R_Calloc(nsqr, double), inf_norm, m1_j/*= (-1)^j */, trshift;
87 R_CheckStack();
88
89 if (n < 1 || Dims[0] != n)
90 error(_("Matrix exponential requires square, non-null matrix"));
91 if(n == 1) {
92 v[0] = exp(v[0]);
93 UNPROTECT(1);
94 return val;
95 }
96
97 /* Preconditioning 1. Shift diagonal by average diagonal if positive. */
98 trshift = 0; /* determine average diagonal element */
99 for (i = 0; i < n; i++) trshift += v[i * np1];
100 trshift /= n;
101 if (trshift > 0.) { /* shift diagonal by -trshift */
102 for (i = 0; i < n; i++) v[i * np1] -= trshift;
103 }
104
105 /* Preconditioning 2. Balancing with dgebal. */
106 F77_CALL(dgebal)("P", &n, v, &n, &ilo, &ihi, perm, &j FCONE);
107 if (j) error(_("dgeMatrix_exp: LAPACK routine dgebal returned %d"), j);
108 F77_CALL(dgebal)("S", &n, v, &n, &ilos, &ihis, scale, &j FCONE);
109 if (j) error(_("dgeMatrix_exp: LAPACK routine dgebal returned %d"), j);
110
111 /* Preconditioning 3. Scaling according to infinity norm */
112 inf_norm = F77_CALL(dlange)("I", &n, &n, v, &n, work FCONE);
113 sqpow = (inf_norm > 0) ? (int) (1 + log(inf_norm)/log(2.)) : 0;
114 if (sqpow < 0) sqpow = 0;
115 if (sqpow > 0) {
116 double scale_factor = 1.0;
117 for (i = 0; i < sqpow; i++) scale_factor *= 2.;
118 for (R_xlen_t i = 0; i < nsqr; i++) v[i] /= scale_factor;
119 }
120
121 /* Pade' approximation. Powers v^8, v^7, ..., v^1 */
122 Matrix_memset(npp, 0, nsqr, sizeof(double));
123 Matrix_memset(dpp, 0, nsqr, sizeof(double));
124 m1_j = -1;
125 for (j = 7; j >=0; j--) {
126 double mult = padec[j];
127 /* npp = m * npp + padec[j] *m */
128 F77_CALL(dgemm)("N", "N", &n, &n, &n, &one, v, &n, npp, &n,
129 &zero, work, &n FCONE FCONE);
130 for (R_xlen_t i = 0; i < nsqr; i++) npp[i] = work[i] + mult * v[i];
131 /* dpp = m * dpp + (m1_j * padec[j]) * m */
132 mult *= m1_j;
133 F77_CALL(dgemm)("N", "N", &n, &n, &n, &one, v, &n, dpp, &n,
134 &zero, work, &n FCONE FCONE);
135 for (R_xlen_t i = 0; i < nsqr; i++) dpp[i] = work[i] + mult * v[i];
136 m1_j *= -1;
137 }
138 /* Zero power */
139 for (R_xlen_t i = 0; i < nsqr; i++) dpp[i] *= -1.;
140 for (j = 0; j < n; j++) {
141 npp[j * np1] += 1.;
142 dpp[j * np1] += 1.;
143 }
144
145 /* Pade' approximation is solve(dpp, npp) */
146 F77_CALL(dgetrf)(&n, &n, dpp, &n, pivot, &j);
147 if (j) error(_("dgeMatrix_exp: dgetrf returned error code %d"), j);
148 F77_CALL(dgetrs)("N", &n, &n, dpp, &n, pivot, npp, &n, &j FCONE);
149 if (j) error(_("dgeMatrix_exp: dgetrs returned error code %d"), j);
150 Memcpy(v, npp, nsqr);
151
152 /* Now undo all of the preconditioning */
153 /* Preconditioning 3: square the result for every power of 2 */
154 while (sqpow--) {
155 F77_CALL(dgemm)("N", "N", &n, &n, &n, &one, v, &n, v, &n,
156 &zero, work, &n FCONE FCONE);
157 Memcpy(v, work, nsqr);
158 }
159 /* Preconditioning 2: apply inverse scaling */
160 for (j = 0; j < n; j++) {
161 R_xlen_t jn = j * n_;
162 for (i = 0; i < n; i++)
163 v[i + jn] *= scale[i]/scale[j];
164 }
165
166
167 /* 2 b) Inverse permutation (if not the identity permutation) */
168 if (ilo != 1 || ihi != n) {
169 /* Martin Maechler's code */
170
171#define SWAP_ROW(I,J) F77_CALL(dswap)(&n, &v[(I)], &n, &v[(J)], &n)
172
173#define SWAP_COL(I,J) F77_CALL(dswap)(&n, &v[(I)*n_], &i1, &v[(J)*n_], &i1)
174
175#define RE_PERMUTE(I) \
176 int p_I = (int) (perm[I]) - 1; \
177 SWAP_COL(I, p_I); \
178 SWAP_ROW(I, p_I)
179
180 /* reversion of "leading permutations" : in reverse order */
181 for (i = (ilo - 1) - 1; i >= 0; i--) {
182 RE_PERMUTE(i);
183 }
184
185 /* reversion of "trailing permutations" : applied in forward order */
186 for (i = (ihi + 1) - 1; i < n; i++) {
187 RE_PERMUTE(i);
188 }
189 }
190
191 /* Preconditioning 1: Trace normalization */
192 if (trshift > 0.) {
193 double mult = exp(trshift);
194 for (R_xlen_t i = 0; i < nsqr; i++) v[i] *= mult;
195 }
196
197 /* Clean up */
198 R_Free(work); R_Free(scale); R_Free(perm); R_Free(npp); R_Free(dpp); R_Free(pivot);
199 UNPROTECT(1);
200 return val;
201}
