264 SEXP inv_permutation(SEXP p_, SEXP zero_p, SEXP zero_res)
265 {
266  int np = 0;
267  if(!isInteger(p_)) {p_ = PROTECT(coerceVector(p_, INTSXP)); np++; }
268  int *p = INTEGER(p_), n = LENGTH(p_);
269  SEXP val = allocVector(INTSXP, n);// (not PROTECT()ing: no alloc from here on)
270  int *v = INTEGER(val), p_0 = asLogical(zero_p), r_0 = asLogical(zero_res);
271  if(!p_0) v--; // ==> use 1-based indices
272  // shorter (but not 100% sure if ok: is LHS always eval'ed *before* RHS ?) :
273  // for(int i=0; i < n; ) v[p[i]] = ++i;
274  for(int i=0; i < n; ) {
275  int j = p[i]; v[j] = (r_0) ? i++ : ++i;
276  }
277  UNPROTECT(np);
278  return val;
279 }
