273 SEXP inv_permutation(SEXP p_, SEXP zero_p, SEXP zero_res)
274 {
275  int np = 0;
276  if(!isInteger(p_)) {p_ = PROTECT(coerceVector(p_, INTSXP)); np++; }
277  int *p = INTEGER(p_), n = LENGTH(p_);
278  SEXP val = allocVector(INTSXP, n);// (not PROTECT()ing: no alloc from here on)
279  int *v = INTEGER(val), p_0 = asLogical(zero_p), r_0 = asLogical(zero_res);
280  if(!p_0) v--; // ==> use 1-based indices
281  // shorter (but not 100% sure if ok: is LHS always eval'ed *before* RHS ?) :
282  // for(int i=0; i < n; ) v[p[i]] = ++i;
283  for(int i=0; i < n; ) {
284  int j = p[i]; v[j] = (r_0) ? i++ : ++i;
285  }
286  UNPROTECT(np);
287  return val;
288 }
