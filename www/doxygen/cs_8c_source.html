510csi *cs_counts (const cs *A, const csi *parent, const csi *post, csi ata)
511{
512 csi i, j, k, n, m, J, s, p, q, jleaf, *ATp, *ATi, *maxfirst, *prevleaf,
513 *ancestor, *head = NULL, *next = NULL, *colcount, *w, *first, *delta ;
514 cs *AT ;
515 if (!CS_CSC (A) || !parent || !post) return (NULL) ; /* check inputs */
516 m = A->m ; n = A->n ;
517 s = 4*n + (ata ? (n+m+1) : 0) ;
518 delta = colcount = cs_malloc (n, sizeof (csi)) ; /* allocate result */
519 w = cs_malloc (s, sizeof (csi)) ; /* get workspace */
520 AT = cs_transpose (A, 0) ; /* AT = A' */
521 if (!AT || !colcount || !w) return (cs_idone (colcount, AT, w, 0)) ;
522 ancestor = w ; maxfirst = w+n ; prevleaf = w+2*n ; first = w+3*n ;
523 for (k = 0 ; k < s ; k++) w [k] = -1 ; /* clear workspace w [0..s-1] */
524 for (k = 0 ; k < n ; k++) /* find first [j] */
525 {
526 j = post [k] ;
527 delta [j] = (first [j] == -1) ? 1 : 0 ; /* delta[j]=1 if j is a leaf */
528 for ( ; j != -1 && first [j] == -1 ; j = parent [j]) first [j] = k ;
529 }
530 ATp = AT->p ; ATi = AT->i ;
531 if (ata) init_ata (AT, post, w, &head, &next) ;
532 for (i = 0 ; i < n ; i++) ancestor [i] = i ; /* each node in its own set */
533 for (k = 0 ; k < n ; k++)
534 {
535 j = post [k] ; /* j is the kth node in postordered etree */
536 if (parent [j] != -1) delta [parent [j]]-- ; /* j is not a root */
537 for (J = HEAD (k,j) ; J != -1 ; J = NEXT (J)) /* J=j for LL'=A case */
538 {
539 for (p = ATp [J] ; p < ATp [J+1] ; p++)
540 {
541 i = ATi [p] ;
542 q = cs_leaf (i, j, first, maxfirst, prevleaf, ancestor, &jleaf);
543 if (jleaf >= 1) delta [j]++ ; /* A(i,j) is in skeleton */
544 if (jleaf == 2) delta [q]-- ; /* account for overlap in q */
545 }
546 }
547 if (parent [j] != -1) ancestor [j] = parent [j] ;
548 }
549 for (j = 0 ; j < n ; j++) /* sum up delta's of each child */
550 {
551 if (parent [j] != -1) colcount [parent [j]] += colcount [j] ;
552 }
553 return (cs_idone (colcount, AT, w, 1)) ; /* success: free workspace */
554}
555/* p [0..n] = cumulative sum of c [0..n-1], and then copy p [0..n-1] into c */
556double cs_cumsum (csi *p, csi *c, csi n)
557{
558 csi i, nz = 0 ;
559 double nz2 = 0 ;
560 if (!p || !c) return (-1) ; /* check inputs */
561 for (i = 0 ; i < n ; i++)
562 {
563 p [i] = nz ;
564 nz += c [i] ;
565 nz2 += c [i] ; /* also in double to avoid csi overflow */
566 c [i] = p [i] ; /* also copy p[0..n-1] back into c[0..n-1]*/
567 }
568 p [n] = nz ;
569 return (nz2) ; /* return sum (c [0..n-1]) */
570}
571/* depth-first-search of the graph of a matrix, starting at node j */
572csi cs_dfs (csi j, cs *G, csi top, csi *xi, csi *pstack, const csi *pinv)
573{
574 csi i, p, p2, done, jnew, head = 0, *Gp, *Gi ;
575 if (!CS_CSC (G) || !xi || !pstack) return (-1) ; /* check inputs */
576 Gp = G->p ; Gi = G->i ;
577 xi [0] = j ; /* initialize the recursion stack */
578 while (head >= 0)
579 {
580 j = xi [head] ; /* get j from the top of the recursion stack */
581 jnew = pinv ? (pinv [j]) : j ;
582 if (!CS_MARKED (Gp, j))
583 {
584 CS_MARK (Gp, j) ; /* mark node j as visited */
585 pstack [head] = (jnew < 0) ? 0 : CS_UNFLIP (Gp [jnew]) ;
586 }
587 done = 1 ; /* node j done if no unvisited neighbors */
588 p2 = (jnew < 0) ? 0 : CS_UNFLIP (Gp [jnew+1]) ;
589 for (p = pstack [head] ; p < p2 ; p++) /* examine all neighbors of j */
590 {
591 i = Gi [p] ; /* consider neighbor node i */
592 if (CS_MARKED (Gp, i)) continue ; /* skip visited node i */
593 pstack [head] = p ; /* pause depth-first search of node j */
594 xi [++head] = i ; /* start dfs at node i */
595 done = 0 ; /* node j is not done */
596 break ; /* break, to start dfs (i) */
597 }
598 if (done) /* depth-first search at node j is done */
599 {
600 head-- ; /* remove j from the recursion stack */
601 xi [--top] = j ; /* and place in the output stack */
602 }
603 }
604 return (top) ;
605}
606/* breadth-first search for coarse decomposition (C0,C1,R1 or R0,R3,C3) */
607static csi cs_bfs (const cs *A, csi n, csi *wi, csi *wj, csi *queue,
608 const csi *imatch, const csi *jmatch, csi mark)
609{
610 csi *Ap, *Ai, head = 0, tail = 0, j, i, p, j2 ;
611 cs *C ;
612 for (j = 0 ; j < n ; j++) /* place all unmatched nodes in queue */
613 {
614 if (imatch [j] >= 0) continue ; /* skip j if matched */
615 wj [j] = 0 ; /* j in set C0 (R0 if transpose) */
616 queue [tail++] = j ; /* place unmatched col j in queue */
617 }
618 if (tail == 0) return (1) ; /* quick return if no unmatched nodes */
619 C = (mark == 1) ? ((cs *) A) : cs_transpose (A, 0) ;
620 if (!C) return (0) ; /* bfs of C=A' to find R3,C3 from R0 */
621 Ap = C->p ; Ai = C->i ;
622 while (head < tail) /* while queue is not empty */
623 {
624 j = queue [head++] ; /* get the head of the queue */
625 for (p = Ap [j] ; p < Ap [j+1] ; p++)
626 {
627 i = Ai [p] ;
628 if (wi [i] >= 0) continue ; /* skip if i is marked */
629 wi [i] = mark ; /* i in set R1 (C3 if transpose) */
630 j2 = jmatch [i] ; /* traverse alternating path to j2 */
631 if (wj [j2] >= 0) continue ;/* skip j2 if it is marked */
632 wj [j2] = mark ; /* j2 in set C1 (R3 if transpose) */
633 queue [tail++] = j2 ; /* add j2 to queue */
634 }
635 }
636 if (mark != 1) cs_spfree (C) ; /* free A' if it was created */
637 return (1) ;
638}

671/* Given A, compute coarse and then fine dmperm */
672csd *cs_dmperm (const cs *A, csi seed)
673{
674 csi m, n, i, j, k, cnz, nc, *jmatch, *imatch, *wi, *wj, *pinv, *Cp, *Ci,
675 *ps, *rs, nb1, nb2, *p, *q, *cc, *rr, *r, *s, ok ;
676 cs *C ;
677 csd *D, *scc ;
678 /* --- Maximum matching ------------------------------------------------- */
679 if (!CS_CSC (A)) return (NULL) ; /* check inputs */
680 m = A->m ; n = A->n ;
681 D = cs_dalloc (m, n) ; /* allocate result */
682 if (!D) return (NULL) ;
683 p = D->p ; q = D->q ; r = D->r ; s = D->s ; cc = D->cc ; rr = D->rr ;
684 jmatch = cs_maxtrans (A, seed) ; /* max transversal */
685 imatch = jmatch + m ; /* imatch = inverse of jmatch */
686 if (!jmatch) return (cs_ddone (D, NULL, jmatch, 0)) ;
687 /* --- Coarse decomposition --------------------------------------------- */
688 wi = r ; wj = s ; /* use r and s as workspace */
689 for (j = 0 ; j < n ; j++) wj [j] = -1 ; /* unmark all cols for bfs */
690 for (i = 0 ; i < m ; i++) wi [i] = -1 ; /* unmark all rows for bfs */
691 cs_bfs (A, n, wi, wj, q, imatch, jmatch, 1) ; /* find C1, R1 from C0*/
692 ok = cs_bfs (A, m, wj, wi, p, jmatch, imatch, 3) ; /* find R3, C3 from R0*/
693 if (!ok) return (cs_ddone (D, NULL, jmatch, 0)) ;
694 cs_unmatched (n, wj, q, cc, 0) ; /* unmatched set C0 */
695 cs_matched (n, wj, imatch, p, q, cc, rr, 1, 1) ; /* set R1 and C1 */
696 cs_matched (n, wj, imatch, p, q, cc, rr, 2, -1) ; /* set R2 and C2 */
697 cs_matched (n, wj, imatch, p, q, cc, rr, 3, 3) ; /* set R3 and C3 */
698 cs_unmatched (m, wi, p, rr, 3) ; /* unmatched set R0 */
699 cs_free (jmatch) ;
700 /* --- Fine decomposition ----------------------------------------------- */
701 pinv = cs_pinv (p, m) ; /* pinv=p' */
702 if (!pinv) return (cs_ddone (D, NULL, NULL, 0)) ;
703 C = cs_permute (A, pinv, q, 0) ;/* C=A(p,q) (it will hold A(R2,C2)) */
704 cs_free (pinv) ;
705 if (!C) return (cs_ddone (D, NULL, NULL, 0)) ;
706 Cp = C->p ;
707 nc = cc [3] - cc [2] ; /* delete cols C0, C1, and C3 from C */
708 if (cc [2] > 0) for (j = cc [2] ; j <= cc [3] ; j++) Cp [j-cc[2]] = Cp [j] ;
709 C->n = nc ;
710 if (rr [2] - rr [1] < m) /* delete rows R0, R1, and R3 from C */
711 {
712 cs_fkeep (C, cs_rprune, rr) ;
713 cnz = Cp [nc] ;
714 Ci = C->i ;
715 if (rr [1] > 0) for (k = 0 ; k < cnz ; k++) Ci [k] -= rr [1] ;
716 }
717 C->m = nc ;
718 scc = cs_scc (C) ; /* find strongly connected components of C*/
719 if (!scc) return (cs_ddone (D, C, NULL, 0)) ;
720 /* --- Combine coarse and fine decompositions --------------------------- */
721 ps = scc->p ; /* C(ps,ps) is the permuted matrix */
722 rs = scc->r ; /* kth block is rs[k]..rs[k+1]-1 */
723 nb1 = scc->nb ; /* # of blocks of A(R2,C2) */
724 for (k = 0 ; k < nc ; k++) wj [k] = q [ps [k] + cc [2]] ;
725 for (k = 0 ; k < nc ; k++) q [k + cc [2]] = wj [k] ;
726 for (k = 0 ; k < nc ; k++) wi [k] = p [ps [k] + rr [1]] ;
727 for (k = 0 ; k < nc ; k++) p [k + rr [1]] = wi [k] ;
728 nb2 = 0 ; /* create the fine block partitions */
729 r [0] = s [0] = 0 ;
730 if (cc [2] > 0) nb2++ ; /* leading coarse block A (R1, [C0 C1]) */
731 for (k = 0 ; k < nb1 ; k++) /* coarse block A (R2,C2) */
732 {
733 r [nb2] = rs [k] + rr [1] ; /* A (R2,C2) splits into nb1 fine blocks */
734 s [nb2] = rs [k] + cc [2] ;
735 nb2++ ;
736 }
737 if (rr [2] < m)
738 {
739 r [nb2] = rr [2] ; /* trailing coarse block A ([R3 R0], C3) */
740 s [nb2] = cc [3] ;
741 nb2++ ;
742 }
743 r [nb2] = m ;
744 s [nb2] = n ;
745 D->nb = nb2 ;
746 cs_dfree (scc) ;
747 return (cs_ddone (D, C, NULL, 1)) ;
748}

810/* find nonzero pattern of Cholesky L(k,1:k-1) using etree and triu(A(:,k)) */
811csi cs_ereach (const cs *A, csi k, const csi *parent, csi *s, csi *w)
812{
813 csi i, p, n, len, top, *Ap, *Ai ;
814 if (!CS_CSC (A) || !parent || !s || !w) return (-1) ; /* check inputs */
815 top = n = A->n ; Ap = A->p ; Ai = A->i ;
816 CS_MARK (w, k) ; /* mark node k as visited */
817 for (p = Ap [k] ; p < Ap [k+1] ; p++)
818 {
819 i = Ai [p] ; /* A(i,k) is nonzero */
820 if (i > k) continue ; /* only use upper triangular part of A */
821 for (len = 0 ; !CS_MARKED (w,i) ; i = parent [i]) /* traverse up etree*/
822 {
823 s [len++] = i ; /* L(k,i) is nonzero */
824 CS_MARK (w, i) ; /* mark i as visited */
825 }
826 while (len > 0) s [--top] = s [--len] ; /* push path onto stack */
827 }
828 for (p = top ; p < n ; p++) CS_MARK (w, s [p]) ; /* unmark all nodes */
829 CS_MARK (w, k) ; /* unmark node k */
830 return (top) ; /* s [top..n-1] contains pattern of L(k,:)*/
831}
832/* compute the etree of A (using triu(A), or A'A without forming A'A */
833csi *cs_etree (const cs *A, csi ata)
834{
835 csi i, k, p, m, n, inext, *Ap, *Ai, *w, *parent, *ancestor, *prev ;
836 if (!CS_CSC (A)) return (NULL) ; /* check inputs */
837 m = A->m ; n = A->n ; Ap = A->p ; Ai = A->i ;
838 parent = cs_malloc (n, sizeof (csi)) ; /* allocate result */
839 w = cs_malloc (n + (ata ? m : 0), sizeof (csi)) ; /* get workspace */
840 if (!w || !parent) return (cs_idone (parent, NULL, w, 0)) ;
841 ancestor = w ; prev = w + n ;
842 if (ata) for (i = 0 ; i < m ; i++) prev [i] = -1 ;
843 for (k = 0 ; k < n ; k++)
844 {
845 parent [k] = -1 ; /* node k has no parent yet */
846 ancestor [k] = -1 ; /* nor does k have an ancestor */
847 for (p = Ap [k] ; p < Ap [k+1] ; p++)
848 {
849 i = ata ? (prev [Ai [p]]) : (Ai [p]) ;
850 for ( ; i != -1 && i < k ; i = inext) /* traverse from i to k */
851 {
852 inext = ancestor [i] ; /* inext = ancestor of i */
853 ancestor [i] = k ; /* path compression */
854 if (inext == -1) parent [i] = k ; /* no anc., parent is k */
855 }
856 if (ata) prev [Ai [p]] = k ;
857 }
858 }
859 return (cs_idone (parent, NULL, w, 1)) ;
860}

901/* apply the ith Householder vector to x */
902csi cs_happly (const cs *V, csi i, double beta, double *x)
903{
904 csi p, *Vp, *Vi ;
905 double *Vx, tau = 0 ;
906 if (!CS_CSC (V) || !x) return (0) ; /* check inputs */
907 Vp = V->p ; Vi = V->i ; Vx = V->x ;
908 for (p = Vp [i] ; p < Vp [i+1] ; p++) /* tau = v'*x */
909 {
910 tau += Vx [p] * x [Vi [p]] ;
911 }
912 tau *= beta ; /* tau = beta*(v'*x) */
913 for (p = Vp [i] ; p < Vp [i+1] ; p++) /* x = x - v*tau */
914 {
915 x [Vi [p]] -= Vx [p] * tau ;
916 }
917 return (1) ;
918}
919/* create a Householder reflection [v,beta,s]=house(x), overwrite x with v,
920 * where (I-beta*v*v')*x = s*e1. See Algo 5.1.1, Golub & Van Loan, 3rd ed. */
921double cs_house (double *x, double *beta, csi n)
922{
923 double s, sigma = 0 ;
924 csi i ;
925 if (!x || !beta) return (-1) ; /* check inputs */
926 for (i = 1 ; i < n ; i++) sigma += x [i] * x [i] ;
927 if (sigma == 0)
928 {
929 s = fabs (x [0]) ; /* s = |x(0)| */
930 (*beta) = (x [0] <= 0) ? 2 : 0 ;
931 x [0] = 1 ;
932 }
933 else
934 {
935 s = sqrt (x [0] * x [0] + sigma) ; /* s = norm (x) */
936 x [0] = (x [0] <= 0) ? (x [0] - s) : (-sigma / (x [0] + s)) ;
937 (*beta) = -1. / (s * x [0]) ;
938 }
939 return (s) ;
940}

949/* consider A(i,j), node j in ith row subtree and return lca(jprev,j) */
950csi cs_leaf (csi i, csi j, const csi *first, csi *maxfirst, csi *prevleaf,
951 csi *ancestor, csi *jleaf)
952{
953 csi q, s, sparent, jprev ;
954 if (!first || !maxfirst || !prevleaf || !ancestor || !jleaf) return (-1) ;
955 *jleaf = 0 ;
956 if (i <= j || first [j] <= maxfirst [i]) return (-1) ; /* j not a leaf */
957 maxfirst [i] = first [j] ; /* update max first[j] seen so far */
958 jprev = prevleaf [i] ; /* jprev = previous leaf of ith subtree */
959 prevleaf [i] = j ;
960 *jleaf = (jprev == -1) ? 1: 2 ; /* j is first or subsequent leaf */
961 if (*jleaf == 1) return (i) ; /* if 1st leaf, q = root of ith subtree */
962 for (q = jprev ; q != ancestor [q] ; q = ancestor [q]) ;
963 for (s = jprev ; s != q ; s = sparent)
964 {
965 sparent = ancestor [s] ; /* path compression */
966 ancestor [s] = q ;
967 }
968 return (q) ; /* q = least common ancester (jprev,j) */
969}

992 {
993 x [j] /= Lx [Lp [j]] ;
994 for (p = Lp [j]+1 ; p < Lp [j+1] ; p++)
995 {
996 x [Li [p]] -= Lx [p] * x [j] ;
997 }
998 }

1018/* [L,U,pinv]=lu(A, [q lnz unz]). lnz and unz can be guess */
1019csn *cs_lu (const cs *A, const css *S, double tol)
1020{
1021 cs *L, *U ;
1022 csn *N ;
1023 double pivot, *Lx, *Ux, *x, a, t ;
1024 csi *Lp, *Li, *Up, *Ui, *pinv, *xi, *q, n, ipiv, k, top, p, i, col, lnz,unz;
1025 if (!CS_CSC (A) || !S) return (NULL) ; /* check inputs */
1026 n = A->n ;
1027 q = S->q ; lnz = S->lnz ; unz = S->unz ;
1028 x = cs_malloc (n, sizeof (double)) ; /* get double workspace */
1029 xi = cs_malloc (2*n, sizeof (csi)) ; /* get csi workspace */
1030 N = cs_calloc (1, sizeof (csn)) ; /* allocate result */
1031 if (!x || !xi || !N) return (cs_ndone (N, NULL, xi, x, 0)) ;
1032 N->L = L = cs_spalloc (n, n, lnz, 1, 0) ; /* allocate result L */
1033 N->U = U = cs_spalloc (n, n, unz, 1, 0) ; /* allocate result U */
1034 N->pinv = pinv = cs_malloc (n, sizeof (csi)) ; /* allocate result pinv */
1035 if (!L || !U || !pinv) return (cs_ndone (N, NULL, xi, x, 0)) ;
1036 Lp = L->p ; Up = U->p ;
1037 for (i = 0 ; i < n ; i++) x [i] = 0 ; /* clear workspace */
1038 for (i = 0 ; i < n ; i++) pinv [i] = -1 ; /* no rows pivotal yet */
1039 for (k = 0 ; k <= n ; k++) Lp [k] = 0 ; /* no cols of L yet */
1040 lnz = unz = 0 ;
1041 for (k = 0 ; k < n ; k++) /* compute L(:,k) and U(:,k) */
1042 {
1043 /* --- Triangular solve --------------------------------------------- */
1044 Lp [k] = lnz ; /* L(:,k) starts here */
1045 Up [k] = unz ; /* U(:,k) starts here */
1046 if ((lnz + n > L->nzmax && !cs_sprealloc (L, 2*L->nzmax + n)) ||
1047 (unz + n > U->nzmax && !cs_sprealloc (U, 2*U->nzmax + n)))
1048 {
1049 return (cs_ndone (N, NULL, xi, x, 0)) ;
1050 }
1051 Li = L->i ; Lx = L->x ; Ui = U->i ; Ux = U->x ;
1052 col = q ? (q [k]) : k ;
1053 top = cs_spsolve (L, A, col, xi, x, pinv, 1) ; /* x = L\A(:,col) */
1054 /* --- Find pivot --------------------------------------------------- */
1055 ipiv = -1 ;
1056 a = -1 ;
1057 for (p = top ; p < n ; p++)
1058 {
1059 i = xi [p] ; /* x(i) is nonzero */
1060 if (pinv [i] < 0) /* row i is not yet pivotal */
1061 {
1062 if ((t = fabs (x [i])) > a)
1063 {
1064 a = t ; /* largest pivot candidate so far */
1065 ipiv = i ;
1066 }
1067 }
1068 else /* x(i) is the entry U(pinv[i],k) */
1069 {
1070 Ui [unz] = pinv [i] ;
1071 Ux [unz++] = x [i] ;
1072 }
1073 }
1074 if (ipiv == -1 || a <= 0) return (cs_ndone (N, NULL, xi, x, 0)) ;
1075 /* tol=1 for partial pivoting; tol<1 gives preference to diagonal */
1076 if (pinv [col] < 0 && fabs (x [col]) >= a*tol) ipiv = col ;
1077 /* --- Divide by pivot ---------------------------------------------- */
1078 pivot = x [ipiv] ; /* the chosen pivot */
1079 Ui [unz] = k ; /* last entry in U(:,k) is U(k,k) */
1080 Ux [unz++] = pivot ;
1081 pinv [ipiv] = k ; /* ipiv is the kth pivot row */
1082 Li [lnz] = ipiv ; /* first entry in L(:,k) is L(k,k) = 1 */
1083 Lx [lnz++] = 1 ;
1084 for (p = top ; p < n ; p++) /* L(k+1:n,k) = x / pivot */
1085 {
1086 i = xi [p] ;
1087 if (pinv [i] < 0) /* x(i) is an entry in L(:,k) */
1088 {
1089 Li [lnz] = i ; /* save unpermuted row in L */
1090 Lx [lnz++] = x [i] / pivot ; /* scale pivot column */
1091 }
1092 x [i] = 0 ; /* x [0..n-1] = 0 for next k */
1093 }
1094 }
1095 /* --- Finalize L and U ------------------------------------------------- */
1096 Lp [n] = lnz ;
1097 Up [n] = unz ;
1098 Li = L->i ; /* fix row indices of L for final pinv */
1099 for (p = 0 ; p < lnz ; p++) Li [p] = pinv [Li [p]] ;
1100 cs_sprealloc (L, 0) ; /* remove extra space from L and U */
1101 cs_sprealloc (U, 0) ;
1102 return (cs_ndone (N, NULL, xi, x, 1)) ; /* success */
1103}

1163/* find an augmenting path starting at column k and extend the match if found */
1164static void cs_augment (csi k, const cs *A, csi *jmatch, csi *cheap, csi *w,
1165 csi *js, csi *is, csi *ps)
1166{
1167 csi found = 0, p, i = -1, *Ap = A->p, *Ai = A->i, head = 0, j ;
1168 js [0] = k ; /* start with just node k in jstack */
1169 while (head >= 0)
1170 {
1171 /* --- Start (or continue) depth-first-search at node j ------------- */
1172 j = js [head] ; /* get j from top of jstack */
1173 if (w [j] != k) /* 1st time j visited for kth path */
1174 {
1175 w [j] = k ; /* mark j as visited for kth path */
1176 for (p = cheap [j] ; p < Ap [j+1] && !found ; p++)
1177 {
1178 i = Ai [p] ; /* try a cheap assignment (i,j) */
1179 found = (jmatch [i] == -1) ;
1180 }
1181 cheap [j] = p ; /* start here next time j is traversed*/
1182 if (found)
1183 {
1184 is [head] = i ; /* column j matched with row i */
1185 break ; /* end of augmenting path */
1186 }
1187 ps [head] = Ap [j] ; /* no cheap match: start dfs for j */
1188 }
1189 /* --- Depth-first-search of neighbors of j ------------------------- */
1190 for (p = ps [head] ; p < Ap [j+1] ; p++)
1191 {
1192 i = Ai [p] ; /* consider row i */
1193 if (w [jmatch [i]] == k) continue ; /* skip jmatch [i] if marked */
1194 ps [head] = p + 1 ; /* pause dfs of node j */
1195 is [head] = i ; /* i will be matched with j if found */
1196 js [++head] = jmatch [i] ; /* start dfs at column jmatch [i] */
1197 break ;
1198 }
1199 if (p == Ap [j+1]) head-- ; /* node j is done; pop from stack */
1200 } /* augment the match if path found: */
1201 if (found) for (p = head ; p >= 0 ; p--) jmatch [is [p]] = js [p] ;
1202}

1204/* find a maximum transveral */
1205csi *cs_maxtrans (const cs *A, csi seed) /*[jmatch [0..m-1]; imatch [0..n-1]]*/
1206{
1207 csi i, j, k, n, m, p, n2 = 0, m2 = 0, *Ap, *jimatch, *w, *cheap, *js, *is,
1208 *ps, *Ai, *Cp, *jmatch, *imatch, *q ;
1209 cs *C ;
1210 if (!CS_CSC (A)) return (NULL) ; /* check inputs */
1211 n = A->n ; m = A->m ; Ap = A->p ; Ai = A->i ;
1212 w = jimatch = cs_calloc (m+n, sizeof (csi)) ; /* allocate result */
1213 if (!jimatch) return (NULL) ;
1214 for (k = 0, j = 0 ; j < n ; j++) /* count nonempty rows and columns */
1215 {
1216 n2 += (Ap [j] < Ap [j+1]) ;
1217 for (p = Ap [j] ; p < Ap [j+1] ; p++)
1218 {
1219 w [Ai [p]] = 1 ;
1220 k += (j == Ai [p]) ; /* count entries already on diagonal */
1221 }
1222 }
1223 if (k == CS_MIN (m,n)) /* quick return if diagonal zero-free */
1224 {
1225 jmatch = jimatch ; imatch = jimatch + m ;
1226 for (i = 0 ; i < k ; i++) jmatch [i] = i ;
1227 for ( ; i < m ; i++) jmatch [i] = -1 ;
1228 for (j = 0 ; j < k ; j++) imatch [j] = j ;
1229 for ( ; j < n ; j++) imatch [j] = -1 ;
1230 return (cs_idone (jimatch, NULL, NULL, 1)) ;
1231 }
1232 for (i = 0 ; i < m ; i++) m2 += w [i] ;
1233 C = (m2 < n2) ? cs_transpose (A,0) : ((cs *) A) ; /* transpose if needed */
1234 if (!C) return (cs_idone (jimatch, (m2 < n2) ? C : NULL, NULL, 0)) ;
1235 n = C->n ; m = C->m ; Cp = C->p ;
1236 jmatch = (m2 < n2) ? jimatch + n : jimatch ;
1237 imatch = (m2 < n2) ? jimatch : jimatch + m ;
1238 w = cs_malloc (5*n, sizeof (csi)) ; /* get workspace */
1239 if (!w) return (cs_idone (jimatch, (m2 < n2) ? C : NULL, w, 0)) ;
1240 cheap = w + n ; js = w + 2*n ; is = w + 3*n ; ps = w + 4*n ;
1241 for (j = 0 ; j < n ; j++) cheap [j] = Cp [j] ; /* for cheap assignment */
1242 for (j = 0 ; j < n ; j++) w [j] = -1 ; /* all columns unflagged */
1243 for (i = 0 ; i < m ; i++) jmatch [i] = -1 ; /* nothing matched yet */
1244 q = cs_randperm (n, seed) ; /* q = random permutation */
1245 for (k = 0 ; k < n ; k++) /* augment, starting at column q[k] */
1246 {
1247 cs_augment (q ? q [k]: k, C, jmatch, cheap, w, js, is, ps) ;
1248 }
1249 cs_free (q) ;
1250 for (j = 0 ; j < n ; j++) imatch [j] = -1 ; /* find row match */
1251 for (i = 0 ; i < m ; i++) if (jmatch [i] >= 0) imatch [jmatch [i]] = i ;
1252 return (cs_idone (jimatch, (m2 < n2) ? C : NULL, w, 1)) ;
1253}

1263{

1345/* post order a forest */
1346csi *cs_post (const csi *parent, csi n)
1347{

1354 for (j = 0 ; j < n ; j++) head [j] = -1 ; /* empty linked lists */
1355 for (j = n-1 ; j >= 0 ; j--) /* traverse nodes in reverse order*/
1356 {

1358 next [j] = head [parent [j]] ; /* add j to list of its parent */
1359 head [parent [j]] = j ;

1362 {

1364 k = cs_tdfs (j, k, head, next, post, stack) ;
1365 }
1366 return (cs_idone (post, NULL, w, 1)) ; /* success; free w, return post */

1414/* sparse QR factorization [V,beta,pinv,R] = qr (A) */
1415csn *cs_qr (const cs *A, const css *S)
1416{
1417 double *Rx, *Vx, *Ax, *x, *Beta ;
1418 csi i, k, p, n, vnz, p1, top, m2, len, col, rnz, *s, *leftmost, *Ap, *Ai,
1419 *parent, *Rp, *Ri, *Vp, *Vi, *w, *pinv, *q ;
1420 cs *R, *V ;
1421 csn *N ; // the result
1422 if (!CS_CSC (A) || !S) return (NULL) ;
1423 n = A->n ; Ap = A->p ; Ai = A->i ; Ax = A->x ;
1424 q = S->q ; parent = S->parent ; pinv = S->pinv ; m2 = S->m2 ;
1425 vnz = S->lnz ; rnz = S->unz ; leftmost = S->leftmost ;
1426 w = cs_malloc (m2+n, sizeof (csi)) ; /* get csi workspace */
1427 x = cs_malloc (m2, sizeof (double)) ; /* get double workspace */
1428 N = cs_calloc (1, sizeof (csn)) ; /* allocate result */
1429 if (!w || !x || !N) return (cs_ndone (N, NULL, w, x, 0)) ;
1430 s = w + m2 ; /* s is size n */
1431 for (k = 0 ; k < m2 ; k++) x [k] = 0 ; /* clear workspace x */
1432 N->L = V = cs_spalloc (m2, n, vnz, 1, 0) ; /* allocate result V */
1433 N->U = R = cs_spalloc (m2, n, rnz, 1, 0) ; /* allocate result R */
1434 N->B = Beta = cs_malloc (n, sizeof (double)) ; /* allocate result Beta */
1435 if (!R || !V || !Beta) return (cs_ndone (N, NULL, w, x, 0)) ;
1436 Rp = R->p ; Ri = R->i ; Rx = R->x ;
1437 Vp = V->p ; Vi = V->i ; Vx = V->x ;
1438 for (i = 0 ; i < m2 ; i++) w [i] = -1 ; /* clear w, to mark nodes */
1439 rnz = 0 ; vnz = 0 ;
1440 for (k = 0 ; k < n ; k++) /* compute V and R */
1441 {
1442 Rp [k] = rnz ; /* R(:,k) starts here */
1443 Vp [k] = p1 = vnz ; /* V(:,k) starts here */
1444 w [k] = k ; /* add V(k,k) to pattern of V */
1445 Vi [vnz++] = k ;
1446 top = n ;
1447 col = q ? q [k] : k ;
1448 for (p = Ap [col] ; p < Ap [col+1] ; p++) /* find R(:,k) pattern */
1449 {
1450 i = leftmost [Ai [p]] ; /* i = min(find(A(i,q))) */
1451 for (len = 0 ; w [i] != k ; i = parent [i]) /* traverse up to k */
1452 {
1453 s [len++] = i ;
1454 w [i] = k ;
1455 }
1456 while (len > 0) s [--top] = s [--len] ; /* push path on stack */
1457 i = pinv [Ai [p]] ; /* i = permuted row of A(:,col) */
1458 x [i] = Ax [p] ; /* x (i) = A(:,col) */
1459 if (i > k && w [i] < k) /* pattern of V(:,k) = x (k+1:m) */
1460 {
1461 Vi [vnz++] = i ; /* add i to pattern of V(:,k) */
1462 w [i] = k ;
1463 }
1464 }
1465 for (p = top ; p < n ; p++) /* for each i in pattern of R(:,k) */
1466 {
1467 i = s [p] ; /* R(i,k) is nonzero */
1468 cs_happly (V, i, Beta [i], x) ; /* apply (V(i),Beta(i)) to x */
1469 Ri [rnz] = i ; /* R(i,k) = x(i) */
1470 Rx [rnz++] = x [i] ;
1471 x [i] = 0 ;
1472 if (parent [i] == k) vnz = cs_scatter (V, i, 0, w, NULL, k, V, vnz);
1473 }
1474 for (p = p1 ; p < vnz ; p++) /* gather V(:,k) = x */
1475 {
1476 Vx [p] = x [Vi [p]] ;
1477 x [Vi [p]] = 0 ;
1478 }
1479 Ri [rnz] = k ; /* R(k,k) = norm (x) */
1480 Rx [rnz++] = cs_house (Vx+p1, Beta+k, vnz-p1) ; /* [v,beta]=house(x) */
1481 }
1482 Rp [n] = rnz ; /* finalize R */
1483 Vp [n] = vnz ; /* finalize V */
1484 return (cs_ndone (N, NULL, w, x, 1)) ; /* success */
1485}

1645/* ordering and symbolic analysis for a Cholesky factorization */

1647{

1670/* solve Gx=b(:,k), where G is either upper (lo=0) or lower (lo=1) triangular */
1671csi cs_spsolve (cs *G, const cs *B, csi k, csi *xi, double *x, const csi *pinv,
1672 csi lo)
1673{

1675 double *Gx, *Bx ;
1676 if (!CS_CSC (G) || !CS_CSC (B) || !xi || !x) return (-1) ;
1677 Gp = G->p ; Gi = G->i ; Gx = G->x ; n = G->n ;
1678 Bp = B->p ; Bi = B->i ; Bx = B->x ;
1679 top = cs_reach (G, B, k, xi, pinv) ; /* xi[top..n-1]=Reach(B(:,k)) */
1680 for (p = top ; p < n ; p++) x [xi [p]] = 0 ; /* clear x */
1681 for (p = Bp [k] ; p < Bp [k+1] ; p++) x [Bi [p]] = Bx [p] ; /* scatter B */
1682 for (px = top ; px < n ; px++)
1683 {
1684 j = xi [px] ; /* x(j) is nonzero */
1685 J = pinv ? (pinv [j]) : j ; /* j maps to col J of G */
1686 if (J < 0) continue ; /* column J is empty */
1687 x [j] /= Gx [lo ? (Gp [J]) : (Gp [J+1]-1)] ;/* x(j) /= G(j,j) */
1688 p = lo ? (Gp [J]+1) : (Gp [J]) ; /* lo: L(j,j) 1st entry */
1689 q = lo ? (Gp [J+1]) : (Gp [J+1]-1) ; /* up: U(j,j) last entry */
1690 for ( ; p < q ; p++)
1691 {
1692 x [Gi [p]] -= Gx [p] * x [j] ; /* x(i) -= G(i,j) * x(j) */
1693 }
1694 }
1695 return (top) ; /* return top of stack */
1696}
1697/* compute nnz(V) = S->lnz, S->pinv, S->leftmost, S->m2 from A and S->parent */
1698static csi cs_vcount (const cs *A, css *S)
1699{
1700 csi i, k, p, pa, n = A->n, m = A->m, *Ap = A->p, *Ai = A->i, *next, *head,
1701 *tail, *nque, *pinv, *leftmost, *w, *parent = S->parent ;
1702 S->pinv = pinv = cs_malloc (m+n, sizeof (csi)) ; /* allocate pinv, */
1703 S->leftmost = leftmost = cs_malloc (m, sizeof (csi)) ; /* and leftmost */
1704 w = cs_malloc (m+3*n, sizeof (csi)) ; /* get workspace */
1705 if (!pinv || !w || !leftmost)
1706 {
1707 cs_free (w) ; /* pinv and leftmost freed later */
1708 return (0) ; /* out of memory */
1709 }
1710 next = w ; head = w + m ; tail = w + m + n ; nque = w + m + 2*n ;
1711 for (k = 0 ; k < n ; k++) head [k] = -1 ; /* queue k is empty */
1712 for (k = 0 ; k < n ; k++) tail [k] = -1 ;
1713 for (k = 0 ; k < n ; k++) nque [k] = 0 ;
1714 for (i = 0 ; i < m ; i++) leftmost [i] = -1 ;
1715 for (k = n-1 ; k >= 0 ; k--)
1716 {
1717 for (p = Ap [k] ; p < Ap [k+1] ; p++)
1718 {
1719 leftmost [Ai [p]] = k ; /* leftmost[i] = min(find(A(i,:)))*/
1720 }
1721 }
1722 for (i = m-1 ; i >= 0 ; i--) /* scan rows in reverse order */
1723 {
1724 pinv [i] = -1 ; /* row i is not yet ordered */
1725 k = leftmost [i] ;
1726 if (k == -1) continue ; /* row i is empty */
1727 if (nque [k]++ == 0) tail [k] = i ; /* first row in queue k */
1728 next [i] = head [k] ; /* put i at head of queue k */
1729 head [k] = i ;
1730 }
1731 S->lnz = 0 ;
1732 S->m2 = m ;
1733 for (k = 0 ; k < n ; k++) /* find row permutation and nnz(V)*/
1734 {
1735 i = head [k] ; /* remove row i from queue k */
1736 S->lnz++ ; /* count V(k,k) as nonzero */
1737 if (i < 0) i = S->m2++ ; /* add a fictitious row */
1738 pinv [i] = k ; /* associate row i with V(:,k) */
1739 if (--nque [k] <= 0) continue ; /* skip if V(k+1:m,k) is empty */
1740 S->lnz += nque [k] ; /* nque [k] is nnz (V(k+1:m,k)) */
1741 if ((pa = parent [k]) != -1) /* move all rows to parent of k */
1742 {
1743 if (nque [pa] == 0) tail [pa] = tail [k] ;
1744 next [tail [k]] = head [pa] ;
1745 head [pa] = next [i] ;
1746 nque [pa] += nque [k] ;
1747 }
1748 }
1749 for (i = 0 ; i < m ; i++) if (pinv [i] < 0) pinv [i] = k++ ;
1750 cs_free (w) ;
1751 return (1) ;
1752}

1783/* C = A(p,p) where A and C are symmetric the upper part stored; pinv not p */
1784cs *cs_symperm (const cs *A, const csi *pinv, csi values)
1785{

1787 double *Cx, *Ax ;
1788 cs *C ;
1789 if (!CS_CSC (A)) return (NULL) ; /* check inputs */
1790 n = A->n ; Ap = A->p ; Ai = A->i ; Ax = A->x ;

1793 if (!C || !w) return (cs_done (C, w, NULL, 0)) ; /* out of memory */
1794 Cp = C->p ; Ci = C->i ; Cx = C->x ;
1795 for (j = 0 ; j < n ; j++) /* count entries in each column of C */
1796 {
1797 j2 = pinv ? pinv [j] : j ; /* column j of A is column j2 of C */
1798 for (p = Ap [j] ; p < Ap [j+1] ; p++)
1799 {
1800 i = Ai [p] ;
1801 if (i > j) continue ; /* skip lower triangular part of A */
1802 i2 = pinv ? pinv [i] : i ; /* row i of A is row i2 of C */
1803 w [CS_MAX (i2, j2)]++ ; /* column count of C */
1804 }
1805 }
1806 cs_cumsum (Cp, w, n) ; /* compute column pointers of C */
1807 for (j = 0 ; j < n ; j++)
1808 {
1809 j2 = pinv ? pinv [j] : j ; /* column j of A is column j2 of C */
1810 for (p = Ap [j] ; p < Ap [j+1] ; p++)
1811 {

1817 }
1818 }
1819 return (cs_done (C, w, NULL, 1)) ; /* success; free workspace, return C */
1820}
1821/* depth-first search and postorder of a tree rooted at node j */

1826 stack [0] = j ; /* place j on the stack */
1827 while (top >= 0) /* while (stack is not empty) */
1828 {
1829 p = stack [top] ; /* p = top of stack */
1830 i = head [p] ; /* i = youngest child of p */

1832 {
1833 top-- ; /* p has no unordered children left */

1835 }
1836 else
1837 {
1838 head [p] = next [i] ; /* remove i from children of p */
1839 stack [++top] = i ; /* start dfs on child node i */
1840 }
1841 }

1868/* sparse Cholesky update/downdate, L*L' + sigma*w*w' (sigma = +1 or -1) */
1869csi cs_updown (cs *L, csi sigma, const cs *C, const csi *parent)
1870{

1872 double *Lx, *Cx, alpha, beta = 1, delta, gamma, w1, w2, *w, beta2 = 1 ;
1873 if (!CS_CSC (L) || !CS_CSC (C) || !parent) return (0) ; /* check inputs */
1874 Lp = L->p ; Li = L->i ; Lx = L->x ; n = L->n ;
1875 Cp = C->p ; Ci = C->i ; Cx = C->x ;
1876 if ((p = Cp [0]) >= Cp [1]) return (1) ; /* return if C empty */
1877 w = cs_malloc (n, sizeof (double)) ; /* get workspace */
1878 if (!w) return (0) ; /* out of memory */
1879 f = Ci [p] ;
1880 for ( ; p < Cp [1] ; p++) f = CS_MIN (f, Ci [p]) ; /* f = min (find (C)) */
