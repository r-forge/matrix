247  SEXP islot = GET_SLOT(x, Matrix_iSym);
248

252  memset(ans, 0, sizeof(cholmod_sparse)); /* zero the struct */
253 
254  ans->itype = CHOLMOD_INT; /* characteristics of the system */
255  ans->dtype = CHOLMOD_DOUBLE;
256  ans->packed = TRUE;
257  /* slots always present */
258  ans->i = INTEGER(islot);
259  ans->p = INTEGER(GET_SLOT(x, Matrix_pSym));
260  /* dimensions and nzmax */
261  ans->nrow = dims[0];
262  ans->ncol = dims[1];
263  /* Allow for over-allocation of the i and x slots. Needed for

265  * check for the length of the x slot, because of the xpt
266  * utility, but the lengths of x and i should agree. */
267  ans->nzmax = LENGTH(islot);
268  /* values depending on ctype */
269  ans->x = xpt(ctype, x);
270  ans->stype = stype(ctype, x);
271  ans->xtype = xtype(ctype);
272

274  ans->sorted = check_sorted_chm(ans);
275  if (!(ans->sorted)) { /* sort columns */
276  if(sort_in_place) {
277  if (!cholmod_sort(ans, &c))
278  error(_("in_place cholmod_sort returned an error code"));
279  ans->sorted = 1;
280  }
281  else {
282  CHM_SP tmp = cholmod_copy_sparse(ans, &c);
283  if (!cholmod_sort(tmp, &c))
284  error(_("cholmod_sort returned an error code"));
285 
286 #ifdef DEBUG_Matrix
287  /* This "triggers" exactly for return values of dtCMatrix_sparse_solve():*/
288  /* Don't want to translate this: want it report */
289  Rprintf("Note: as_cholmod_sparse() needed cholmod_sort()ing\n");
290 #endif
291  chm2Ralloc(ans, tmp);
292  cholmod_free_sparse(&tmp, &c);
293  }
294  }
295 
296  if (check_Udiag && ctype % 3 == 2 // triangular
297  && (*diag_P(x) == 'U')) { /* diagU2N(.) "in place" : */
298  double one[] = {1, 0};
299  CHM_SP eye = cholmod_speye(ans->nrow, ans->ncol, ans->xtype, &c);
300  CHM_SP tmp = cholmod_add(ans, eye, one, one, TRUE, TRUE, &c);
301 
302 #ifdef DEBUG_Matrix_verbose /* quite often, e.g. in ../tests/indexing.R */
303  Rprintf("Note: as_cholmod_sparse(<ctype=%d>) - diagU2N\n", ctype);
304 #endif
305  chm2Ralloc(ans, tmp);
306  cholmod_free_sparse(&tmp, &c);
307  cholmod_free_sparse(&eye, &c);
308  } /* else :
309  * NOTE: if(*diag_P(x) == 'U'), the diagonal is lost (!);
310  * ---- that may be ok, e.g. if we are just converting from/to Tsparse,
311  * but is *not* at all ok, e.g. when used before matrix products */
312 
313  return ans;
314 }

399  }

1187 void chm_diagN2U(CHM_SP chx, int uploT, Rboolean do_realloc)
1188 {
1189  int i, n = chx->nrow, nnz = (int)cholmod_nnz(chx, &c),
1190  n_nnz = nnz - n, /* the new nnz : we will have removed n entries */
1191  i_to = 0, i_from = 0;
1192 
1193  if(chx->ncol != n)
1194  error(_("chm_diagN2U(<non-square matrix>): nrow=%d, ncol=%d"),
1195  n, chx->ncol);
1196 
1197  if (!chx->sorted || !chx->packed) cholmod_sort(chx, &c);
1198  /* dimensions and nzmax */
1199 
1200 #define _i(I) ( (int*) chx->i)[I]
1201 #define _x(I) ((double*) chx->x)[I]
1202 #define _p(I) ( (int*) chx->p)[I]
1203 
1204  /* work by copying from i_from to i_to ==> MUST i_to <= i_from */
1205 
1206  if(uploT == 1) { /* "U" : upper triangular */
1207 
1208  for(i = 0; i < n; i++) { /* looking at i-th column */
1209  int j, n_i = _p(i+1) - _p(i); /* = #{entries} in this column */
1210 
1211  /* 1) copy all but the last _above-diagonal_ column-entries: */
1212  for(j = 1; j < n_i; j++, i_to++, i_from++) {
1213  _i(i_to) = _i(i_from);
1214  _x(i_to) = _x(i_from);
1215  }
1216 
1217  /* 2) drop the last column-entry == diagonal entry */
1218  i_from++;
1219  }
1220  }
1221  else if(uploT == -1) { /* "L" : lower triangular */
