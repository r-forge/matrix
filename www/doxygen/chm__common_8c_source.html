5SEXP checkpi(SEXP p, SEXP i, int m, int n)
6{
7
8#define MKMS(_FORMAT_, ...) mkString(Matrix_sprintf(_FORMAT_, __VA_ARGS__))
9
10 if (TYPEOF(p) != INTSXP)
11 return MKMS(_("'%s' slot is not of type \"%s\""),
12 "p", "integer");
13 if (XLENGTH(p) - 1 != n)
14 return MKMS(_("'%s' slot does not have length %s"),
15 "p", "Dim[2]+1");
16 int *pp = INTEGER(p);
17 if (pp[0] != 0)
18 return MKMS(_("first element of '%s' slot is not 0"),
19 "p");
20 int j;
21 for (j = 1; j <= n; ++j) {
22 if (pp[j] == NA_INTEGER)
23 return MKMS(_("'%s' slot contains NA"),
24 "p");
25 if (pp[j] < pp[j - 1])
26 return MKMS(_("'%s' slot is not nondecreasing"),
27 "p");
28 if (pp[j] - pp[j - 1] > m)
29 return MKMS(_("first differences of '%s' slot exceed %s"),
30 "p", "Dim[1]");
31 }
32
33 if (TYPEOF(i) != INTSXP)
34 return MKMS(_("'%s' slot is not of type \"%s\""),
35 "i", "integer");
36 if (XLENGTH(i) < pp[n])
37 return MKMS(_("'%s' slot has length less than %s"),
38 "i", "p[length(p)]");
39 int *pi = INTEGER(i), k, kend, ik, i0, sorted = 1;
40 for (j = 1, k = 0; j <= n; ++j) {
41 kend = pp[j];
42 i0 = -1;
43 while (k < kend) {
44 ik = pi[k];
45 if (ik == NA_INTEGER)
46 return MKMS(_("'%s' slot contains NA"),
47 "i");
48 if (ik < 0 || ik >= m)
49 return MKMS(_("'%s' slot has elements not in {%s}"),
50 "i", "0,...,Dim[1]-1");
51 if (ik < i0)
52 sorted = 0;
53 else if (ik == i0)
54 return MKMS(_("'%s' slot is not increasing within columns after sorting"),
55 "i");
56 i0 = ik;
57 ++k;
58 }
59 }
60
61 SEXP ans = allocVector(LGLSXP, 1);
62 LOGICAL(ans)[0] = sorted;
63 return ans;
64}

741 UNPROTECT(3);

758double cholmod_factor_ldetA(cholmod_factor *L)
759{
760 int i, j, p;
761 double ans = 0;
762 if (L->is_super) {
763 int *lpi = (int *) L->pi, *lsup = (int *) L->super;
764 for (i = 0; i < L->nsuper; i++) {
765 int nrp1 = 1 + lpi[i + 1] - lpi[i],
766 nc = lsup[i + 1] - lsup[i];
767 double *x = (double *) L->x + ((int *) L->px)[i];
768 for (R_xlen_t jn = 0, j = 0; j < nc; j++, jn += nrp1)
769 ans += 2.0 * log(fabs(x[jn]));
770 }
771 } else {
772 int *li = (int *) L->i, *lp = (int *) L->p;
773 double *lx = (double *) L->x;
774 for (j = 0; j < L->n; j++) {
775 for (p = lp[j]; li[p] != j && p < lp[j + 1]; p++)
776 ;
777 if (li[p] != j) {
778 error(_("invalid simplicial Cholesky factorization: structural zero on main diagonal in column %d"),
779 j);
780 break;
781 }
782 ans += log(lx[p] * ((L->is_ll) ? lx[p] : 1.0));
783 }
784 }
785 return ans;
786}
