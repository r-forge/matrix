124 // TODO: for other classes consider using a trick as RallocedReal() in ./chm_common.c
125 static const char
126 *valid_cM [] = { // the only ones, for "the moment". FIXME: extend (!)
127 x_CLASSES,
128 ""},
129 // value: assume a "dsparseVector" for now -- slots: (i, length, x)
130 *valid_spv[] = { sparseVECTOR, // = "the one with the same slot-class"
131 // all others: ctype_v slot_kind
132 "nsparseVector",// 1 -1
133 "lsparseVector",// 2 1
134 "isparseVector",// 3 2
135 "dsparseVector",// 4 0
136 "zsparseVector",// 5 3
137 ""};
138
139 int ctype_x = R_check_class_etc(x, valid_cM),
140 ctype_v = R_check_class_etc(value, valid_spv);
141 if (ctype_x < 0)
142 error(_("invalid class of 'x' in Csparse_subassign()"));
143 if (ctype_v < 0)
144 error(_("invalid class of 'value' in Csparse_subassign()"));
145 Rboolean value_is_nsp = ctype_v == 1;
146#ifndef _has_x_slot_ // i.e. "n.CMatrix" : sparseVECTOR == "nsparseVector"
147 if(!value_is_nsp) value_is_nsp = (ctype_v == 0);
148#endif
149
150 SEXP
151 islot = GET_SLOT(x, Matrix_iSym),
152 dimslot = GET_SLOT(x, Matrix_DimSym),
153 i_cp = PROTECT(coerceVector(i_, INTSXP)),
154 j_cp = PROTECT(coerceVector(j_, INTSXP));
155 // for d.CMatrix and l.CMatrix but not n.CMatrix:
156
157 int *dims = INTEGER(dimslot),
158 ncol = dims[1], /* nrow = dims[0], */
159 *i = INTEGER(i_cp), len_i = LENGTH(i_cp),
160 *j = INTEGER(j_cp), len_j = LENGTH(j_cp),
161 k,
162 nnz_x = LENGTH(islot);
163 int nnz = nnz_x;
164
165#define MATRIX_SUBASSIGN_VERBOSE
166// Temporary hack for debugging --- remove eventually -- FIXME
167#ifdef MATRIX_SUBASSIGN_VERBOSE
168 Rboolean verbose = i[0] < 0;
169 if(verbose) {
170 i[0] = -i[0];
171 REprintf("Csparse_subassign() x[i,j] <- val; x is \"%s\"; value \"%s\" is_nsp=%d\n",
172 valid_cM[ctype_x], valid_spv[ctype_v], (int)value_is_nsp);
173 }
174#endif
175
176 SEXP val_i_slot, val_x_slot;
177 val_i_slot = PROTECT(coerceVector(GET_SLOT(value, Matrix_iSym), REALSXP));
178 double *val_i = REAL(val_i_slot);
179 int nnz_val = LENGTH(GET_SLOT(value, Matrix_iSym)), n_prot = 4;
180 Type_x *val_x = NULL;
181 if(!value_is_nsp) {
182 if(ctype_v) { // matrix 'x' and 'value' are of different kinds
183 switch((enum x_slot_kind) slot_kind_x) {
184 case x_pattern:// "n"
185 case x_logical:// "l"
186 if(ctype_v >= 3)
187 warning(_("x[] <- val: val is coerced to logical for \"%s\" x"),
188 valid_cM[ctype_x]);
189 break;
190 case x_integer:
191 if(ctype_v >= 4)
192 error(_("x[] <- val: val should be integer or logical, is coerced to integer, for \"%s\" x"),
193 valid_cM[ctype_x]);
194 break;
195 case x_double:
196 case x_complex: // coercion should be tried (and fail for complex -> double) below
197 break;
198 default:
199 error(_("programming error in Csparse_subassign() should never happen"));
200 }
201 // otherwise: "coerce" : as(., <sparseVector>) :
202 val_x_slot = PROTECT(coerceVector(GET_SLOT(value, Matrix_xSym), SXP_x)); n_prot++;
203 val_x = STYP_x(val_x_slot);
204 } else {
205 val_x = STYP_x( GET_SLOT(value, Matrix_xSym));
206 }
207 }
208 int64_t len_val = (int64_t) asReal(GET_SLOT(value, Matrix_lengthSym));
209 /* llen_i = (int64_t) len_i; */
210
211 SEXP ans;
212 /* Instead of simple "duplicate": PROTECT(ans = duplicate(x)) , build up: */
213 // Assuming that ans will have the same basic Matrix type as x :
214 ans = PROTECT(newObject(valid_cM[ctype_x]));
215 SET_SLOT(ans, Matrix_DimSym, duplicate(dimslot));
216 SET_SLOT(ans, Matrix_DimNamesSym, duplicate(GET_SLOT(x, Matrix_DimNamesSym)));
217 SET_SLOT(ans, Matrix_pSym, duplicate(GET_SLOT(x, Matrix_pSym)));
218 SEXP r_pslot = GET_SLOT(ans, Matrix_pSym);
219 // and assign the i- and x- slots at the end, as they are potentially modified
220 // not just in content, but also in their *length*
221 int *rp = INTEGER(r_pslot),
222 *ri = R_Calloc(nnz_x, int); // to contain the final i - slot
223 Memcpy(ri, INTEGER(islot), nnz_x);
224 Type_x_0_init(z_ans);
225 Type_x_1_init(one_ans);
226#ifdef _has_x_slot_
227 Type_x *rx = R_Calloc(nnz_x, Type_x); // to contain the final x - slot
228 Memcpy(rx, STYP_x(GET_SLOT(x, Matrix_xSym)), nnz_x);
229#endif
230 // NB: nnz_x : will always be the "current allocated length" of (i, x) slots
231 // -- nnz : the current *used* length; always nnz <= nnz_x
232
233 int jj, j_val = 0; // in "running" conceptionally through all value[i+ jj*len_i]
234 // values, we are "below"/"before" the (j_val)-th non-zero one.
235 // e.g. if value = (0,0,...,0), have nnz_val == 0, j_val must remain == 0
236 int64_t ii_val;// == "running" index (i + jj*len_i) % len_val for value[]
237 for(jj = 0, ii_val=0; jj < len_j; jj++) {
238 int j__ = j[jj];
239 /* int64_t j_l = jj * llen_i; */
240 R_CheckUserInterrupt();
241 for(int ii = 0; ii < len_i; ii++, ii_val++) {
242 int i__ = i[ii], p1, p2;
243 if(nnz_val && ii_val >= len_val) { // "recycle" indexing into value[]
244 ii_val -= len_val; // = (ii + jj*len_i) % len_val
245 j_val = 0;
246 }
247 int64_t ii_v1;//= ii_val + 1;
248 Type_x v, /* := value[(ii + j_l) % len_val]
249 = .sparseVector_sub((ii + j_l) % len_val,
250 nnz_val, val_i, val_x, len_val)
251 */
252 M_ij;
253 int ind;
254 Rboolean have_entry = FALSE;
255
256 // note that rp[]'s may have *changed* even when 'j' remained!
257 // "FIXME": do this only *when* rp[] has changed
258 p1 = rp[j__], p2 = rp[j__ + 1];
259
260 // v := value[(ii + j_l) % len_val] = value[ii_val]
261 v = z_ans;
262 if(j_val < nnz_val) { // maybe find v := non-zero value[ii_val]
263 ii_v1 = ii_val + 1;
264 if(ii_v1 < val_i[j_val]) { // typical case: are still in zero-stretch
265 // v = z_ans (== 0)
266 } else if(ii_v1 == val_i[j_val]) { // have a match
267 v = (value_is_nsp) ? one_ans : val_x[j_val];
268 j_val++;// from now on, look at the next non-zero entry
269 } else { // ii_v1 > val_i[j_val]
270 REprintf("programming thinko in Csparse_subassign(*, i=%d,j=%d): ii_v=%d, v@i[j_val=%ld]=%g\n",
271 i__,j__, ii_v1, j_val, val_i[j_val]);
272 j_val++;// from now on, look at the next non-zero entry
273 }
274 }
275 // --------------- M_ij := getM(i., j.) --------------------------------
276 M_ij = z_ans; // as in ./t_sparseVector.c
277 for(ind = p1; ind < p2; ind++) {
278 if(ri[ind] >= i__) {
279 if(ri[ind] == i__) {
280#ifdef _has_x_slot_
281 M_ij = rx[ind];
282#else
283 M_ij = 1;
284#endif
285#ifdef MATRIX_SUBASSIGN_VERBOSE
286 if(verbose)
287 REprintf("have entry x[%d, %d] = %g\n", i__, j__,
288# ifdef _CPLX_x
289 (double)M_ij.r);
290# else
291 (double)M_ij);
292# endif
293#endif
294 have_entry = TRUE;
295 } else { // ri[ind] > i__
296#ifdef MATRIX_SUBASSIGN_VERBOSE
297 if(verbose)
298 REprintf("@i > i__ = %d --> ind-- = %d\n", i__, ind);
299#endif
300 }
301 break;
302 }
303 }
304
305 //-- R: if(getM(i., j.) != (v <- getV(ii, jj)))
306
307 // if(contents differ) ==> value needs to be changed :
308#ifdef _CPLX_x
309 if(M_ij.r != v.r || M_ij.i != v.i) {
310#else
311 if(M_ij != v) {
312#endif
313
314#ifdef MATRIX_SUBASSIGN_VERBOSE
315 if(verbose)
316 REprintf("setting x[%d, %d] <- %g", i__,j__,
317# ifdef _CPLX_x
318 (double) v.r);
319# else
320 (double) v);
321# endif
322#endif
323 // (otherwise: nothing to do):
324 // setM(i__, j__, v)
325 // ----------------------------------------------------------
326
327#ifndef _has_x_slot_
328 if(v == z_ans) {
329 // Case I ----- remove x[i, j] = M_ij which we know is *non*-zero
330// BUT it is more efficient (memory-management!) *NOT* to remove,
332// currently using drop0() in R code
333 // we know : have_entry = TRUE ;
334 // ri[ind] == i__; M_ij = rx[ind];
335#ifdef MATRIX_SUBASSIGN_VERBOSE
336 if(verbose)
337 REprintf(" rm ind=%d\n", ind);
338#endif
339 // remove the 'ind'-th element from x@i and x@x :
340 nnz-- ;
341 for(k=ind; k < nnz; k++) {
342 ri[k] = ri[k+1];
343#ifdef _has_x_slot_
344 rx[k] = rx[k+1];
345#endif
346 }
347 for(k=j__ + 1; k <= ncol; k++) {
348 rp[k] = rp[k] - 1;
349 }
350 }
351 else
352#endif
353 if(have_entry) {
354 // Case II ----- replace (non-empty) x[i,j] by v -------
355#ifdef MATRIX_SUBASSIGN_VERBOSE
356 if(verbose)
357 REprintf(" repl. ind=%d\n", ind);
358#endif
359#ifdef _has_x_slot_
360 rx[ind] = v;
361#endif
362 } else {
363 // Case III ----- v != 0 : insert v into "empty" x[i,j] ----
364
365 // extend the i and x slot by one entry : ---------------------
366 if(nnz+1 > nnz_x) { // need to reallocate:
367#ifdef MATRIX_SUBASSIGN_VERBOSE
368 if(verbose) REprintf(" R_Realloc()ing: nnz_x=%d", nnz_x);
369#endif
370 // do it "only" 1x,..4x at the very most increasing by the
371 // nnz-length of "value":
372 nnz_x += (1 + nnz_val / 4);
373#ifdef MATRIX_SUBASSIGN_VERBOSE
374 if(verbose) REprintf("(nnz_v=%d) --> %d ", nnz_val, nnz_x);
375#endif
376 // C doc on realloc() says that the old content is *preserve*d
377 ri = R_Realloc(ri, nnz_x, int);
378#ifdef _has_x_slot_
379 rx = R_Realloc(rx, nnz_x, Type_x);
380#endif
381 }
382 // 3) fill them ...
383
384 int i1 = ind;
385#ifdef MATRIX_SUBASSIGN_VERBOSE
386 if(verbose)
387 REprintf(" INSERT p12=(%d,%d) -> ind=%d -> i1 = %d\n",
388 p1,p2, ind, i1);
389#endif
390
391 // shift the "upper values" *before* the insertion:
392 for(int l = nnz-1; l >= i1; l--) {
393 ri[l+1] = ri[l];
394#ifdef _has_x_slot_
395 rx[l+1] = rx[l];
396#endif
397 }
398 ri[i1] = i__;
399#ifdef _has_x_slot_
400 rx[i1] = v;
401#endif
402 nnz++;
403 // the columns j "right" of the current one :
404 for(k=j__ + 1; k <= ncol; k++)
405 rp[k]++;
406 }
407 }
408#ifdef MATRIX_SUBASSIGN_VERBOSE
409 else if(verbose) REprintf("M_ij == v = %g\n",
410# ifdef _CPLX_x
411 (double) v.r);
412# else
413 (double) v);
414# endif
415#endif
416 }// for( ii )
417 }// for( jj )
418
419 if(ctype_x == 1) { // triangularMatrix: copy the 'diag' and 'uplo' slots
420 SET_SLOT(ans, Matrix_uploSym, duplicate(GET_SLOT(x, Matrix_uploSym)));
421 SET_SLOT(ans, Matrix_diagSym, duplicate(GET_SLOT(x, Matrix_diagSym)));
422 }
423 // now assign the i- and x- slots, free memory and return :
424 PROTECT(islot = allocVector(INTSXP, nnz));
425 Memcpy(INTEGER(islot), ri, nnz);
426 SET_SLOT(ans, Matrix_iSym, islot);
427 UNPROTECT(1);
428#ifdef _has_x_slot_
429 PROTECT(islot = allocVector(SXP_x, nnz));
430 Memcpy(STYP_x(islot), rx, nnz);
431 SET_SLOT(ans, Matrix_xSym, islot);
432 UNPROTECT(1);
433 R_Free(rx);
434#endif
435 R_Free(ri);
436 UNPROTECT(n_prot);
437 return ans;
438}
