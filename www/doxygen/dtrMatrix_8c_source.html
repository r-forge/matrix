105  int rt = asLogical(right); /* if(rt), compute b %*% op(a), else op(a) %*% b */
106  int tr = asLogical(trans);/* if true, use t(a) */
107  int *adims = INTEGER(GET_SLOT(a, Matrix_DimSym)),
108  *bdims = INTEGER(GET_SLOT(val, Matrix_DimSym));

110  double one = 1.;
111 
112  if (adims[0] != adims[1])
113  error(_("dtrMatrix must be square"));
114  if ((rt && adims[0] != n) || (!rt && adims[1] != m))
115  error(_("Matrices are not conformable for multiplication"));
116  if (m < 1 || n < 1) {
117 /* error(_("Matrices with zero extents cannot be multiplied")); */
118  } else /* BLAS */

122  REAL(GET_SLOT(a, Matrix_xSym)), adims,

128 SEXP dtrMatrix_dtrMatrix_mm(SEXP a, SEXP b, SEXP right, SEXP trans)
129 {
130  /* to be called from "%*%" and crossprod(), tcrossprod(),
131  * from ../R/products.R
132  *
133  * TWO cases : (1) result is triangular <=> uplo are equal
134  * === (2) result is "general"
135  */
136  SEXP val,/* = in case (2): PROTECT(dup_mMatrix_as_dgeMatrix(b)); */
137  d_a = GET_SLOT(a, Matrix_DimSym),
138  uplo_a = GET_SLOT(a, Matrix_uploSym),
139  diag_a = GET_SLOT(a, Matrix_diagSym);
140  /* if(rt), compute b %*% a, else a %*% b */
141  int rt = asLogical(right);
142  int tr = asLogical(trans);/* if true, use t(a) */
143  int *adims = INTEGER(d_a), n = adims[0];
144  double *valx = (double *) NULL /*Wall*/;
145  const char
146  *uplo_a_ch = CHAR(STRING_ELT(uplo_a, 0)), /* = uplo_P(a) */
147  *diag_a_ch = CHAR(STRING_ELT(diag_a, 0)); /* = diag_P(a) */
148  Rboolean same_uplo = (*uplo_a_ch == *uplo_P(b)),
149  uDiag_b = /* -Wall: */ FALSE;
150 
151  if (INTEGER(GET_SLOT(b, Matrix_DimSym))[0] != n)
152  /* validity checking already "assures" square matrices ... */
153  error(_("\"dtrMatrix\" objects in '%*%' must have matching (square) dimension"));
154  if(same_uplo) {
155  /* ==> result is triangular -- "dtrMatrix" !
156  * val := dup_mMatrix_as_dtrMatrix(b) : */
157  int sz = n * n;
158  val = PROTECT(NEW_OBJECT(MAKE_CLASS("dtrMatrix")));
159  SET_SLOT(val, Matrix_uploSym, duplicate(uplo_a));
160  SET_SLOT(val, Matrix_DimSym, duplicate(d_a));
161  SET_DimNames(val, b);
162  valx = REAL(ALLOC_SLOT(val, Matrix_xSym, REALSXP, sz));
163  Memcpy(valx, REAL(GET_SLOT(b, Matrix_xSym)), sz);
164  if((uDiag_b = *diag_P(b) == 'U')) {
165  /* unit-diagonal b - may contain garbage in diagonal */
166  for (int i = 0; i < n; i++)
167  valx[i * (n+1)] = 1.;
168  }
169  } else { /* different "uplo" ==> result is "dgeMatrix" ! */
170  val = PROTECT(dup_mMatrix_as_dgeMatrix(b));
171  }
172  if (n >= 1) {
173  double alpha = 1.;
174  /* Level 3 BLAS - DTRMM(): Compute one of the matrix multiplication operations
175  * B := alpha*op( A )*B ["L"], or B := alpha*B*op( A ) ["R"],
176  * where trans_A determines op(A):= A "N"one or
177  * op(A):= t(A) "T"ransposed */
178  F77_CALL(dtrmm)(rt ? "R" : "L", uplo_a_ch,
179  /*trans_A = */ tr ? "T" : "N", diag_a_ch, &n, &n, &alpha,
180  REAL(GET_SLOT(a, Matrix_xSym)), adims,
181  REAL(GET_SLOT(val, Matrix_xSym)), &n);
182  }
183  if(same_uplo) {
184  make_d_matrix_triangular(valx, a); /* set "other triangle" to 0 */
185  if(*diag_a_ch == 'U' && uDiag_b) /* result remains uni-diagonal */
186  SET_SLOT(val, Matrix_diagSym, duplicate(diag_a));
187  }
188  UNPROTECT(1);
189  return val;
190 }
